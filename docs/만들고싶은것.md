import androidx.compose.animation.animateColorAsState
import androidx.compose.animation.core.animateFloatAsState

// --- 색상 정의 ---
val Indigo50 = Color(0xFFEEF2FF)
val Indigo100 = Color(0xFFE0E7FF)
val Indigo500 = Color(0xFF6366F1)
val Indigo600 = Color(0xFF4F46E5)
val Indigo900 = Color(0xFF312E81)

val Emerald50 = Color(0xFFECFDF5)
val Emerald100 = Color(0xFFD1FAE5)
val Emerald500 = Color(0xFF10B981)
val Emerald600 = Color(0xFF059669)
val Emerald900 = Color(0xFF064E3B)

val Gray100 = Color(0xFFF3F4F6)
val Gray400 = Color(0xFF9CA3AF)
val Gray500 = Color(0xFF6B7280)
val Gray800 = Color(0xFF1F2937)

@Composable
fun FocusTimerScreen() {
    // --- 상수 정의 ---
    val focusTime = 25 * 60
    val breakTime = 5 * 60

    // --- 상태 관리 ---
    var timeLeft by remember { mutableStateOf(focusTime) }
    var isActive by remember { mutableStateOf(false) }
    var mode by remember { mutableStateOf("focus") } // "focus" or "break"

    // --- 애니메이션용 색상 ---
    val targetBackgroundColor = if (mode == "focus") Indigo50 else Emerald50
    val backgroundColor by animateColorAsState(targetValue = targetBackgroundColor, label = "bgColor")

    val targetPrimaryColor = if (mode == "focus") Indigo500 else Emerald500
    val primaryColor by animateColorAsState(targetValue = targetPrimaryColor, label = "primaryColor")

    val targetTextColor = if (mode == "focus") Indigo900 else Emerald900
    val textColor by animateColorAsState(targetValue = targetTextColor, label = "textColor")

    // --- 타이머 로직 ---
    LaunchedEffect(isActive, timeLeft) {
        if (isActive && timeLeft > 0) {
            delay(1000L)
            timeLeft--
        } else if (timeLeft == 0) {
            // 시간이 다 되면 모드 전환
            isActive = false
            if (mode == "focus") {
                mode = "break"
                timeLeft = breakTime
            } else {
                mode = "focus"
                timeLeft = focusTime
            }
        }
    }

    // --- 헬퍼 함수 ---
    fun formatTime(seconds: Int): String {
        val mins = seconds / 60
        val secs = seconds % 60
        return "%02d:%02d".format(mins, secs)
    }

    fun changeMode(newMode: String) {
        mode = newMode
        timeLeft = if (newMode == "focus") focusTime else breakTime
        isActive = false
    }

    fun resetTimer() {
        isActive = false
        timeLeft = if (mode == "focus") focusTime else breakTime
    }

    // 진행률 계산
    val totalTime = if (mode == "focus") focusTime else breakTime
    val progress = if (totalTime > 0) (totalTime - timeLeft) / totalTime.toFloat() else 0f
    
    // 프로그레스 바 애니메이션
    val animatedProgress by animateFloatAsState(
        targetValue = progress,
        animationSpec = tween(durationMillis = 1000, easing = androidx.compose.animation.core.LinearEasing),
        label = "progress"
    )

    // --- UI 구성 ---
    Scaffold(
        containerColor = backgroundColor
    ) { paddingValues ->
        Column(
            modifier = Modifier
                .fillMaxSize()
                .padding(paddingValues)
                .padding(24.dp),
            horizontalAlignment = Alignment.CenterHorizontally,
            verticalArrangement = Arrangement.SpaceBetween
        ) {
            // 1. 상단 모드 선택 탭
            Box(
                modifier = Modifier
                    .background(Gray100, RoundedCornerShape(50))
                    .padding(6.dp)
            ) {
                Row {
                    ModeButton(
                        text = "집중",
                        icon = Icons.Rounded.SelfImprovement, // Brain 대체
                        isSelected = mode == "focus",
                        selectedColor = Indigo500,
                        onClick = { changeMode("focus") }
                    )
                    Spacer(modifier = Modifier.width(8.dp))
                    ModeButton(
                        text = "휴식",
                        icon = Icons.Rounded.Coffee,
                        isSelected = mode == "break",
                        selectedColor = Emerald500,
                        onClick = { changeMode("break") }
                    )
                }
            }

            // 2. 타이머 디스플레이 (원형 프로그레스)
            Box(
                contentAlignment = Alignment.Center,
                modifier = Modifier.size(300.dp)
            ) {
                // 원형 프로그레스 바 그리기
                CircularProgressBar(
                    progress = animatedProgress,
                    color = primaryColor,
                    strokeWidth = 12.dp,
                    modifier = Modifier.size(280.dp)
                )

                // 중앙 텍스트
                Column(
                    horizontalAlignment = Alignment.CenterHorizontally
                ) {
                    Text(
                        text = formatTime(timeLeft),
                        fontSize = 72.sp,
                        fontWeight = FontWeight.Black,
                        color = textColor,
                        letterSpacing = (-2).sp
                    )
                    
                    Spacer(modifier = Modifier.height(8.dp))
                    
                    Surface(
                        color = if (mode == "focus") Indigo100 else Emerald100,
                        shape = RoundedCornerShape(50),
                    ) {
                        Text(
                            text = if (isActive) (if (mode == "focus") "Focusing" else "Resting") else "Ready",
                            fontSize = 14.sp,
                            fontWeight = FontWeight.Bold,
                            color = primaryColor,
                            modifier = Modifier.padding(horizontal = 16.dp, vertical = 6.dp)
                        )
                    }
                }
            }

            // 3. 컨트롤 버튼 영역
            Column(
                horizontalAlignment = Alignment.CenterHorizontally,
                verticalArrangement = Arrangement.spacedBy(32.dp)
            ) {
                Row(
                    verticalAlignment = Alignment.CenterVertically,
                    horizontalArrangement = Arrangement.Center,
                    modifier = Modifier.fillMaxWidth()
                ) {
                    // 리셋 버튼
                    IconButton(
                        onClick = { resetTimer() },
                        modifier = Modifier
                            .size(56.dp)
                            .background(Gray100, CircleShape)
                    ) {
                        Icon(
                            imageVector = Icons.Rounded.Refresh, // RotateCcw 대체
                            contentDescription = "Reset",
                            tint = Gray500,
                            modifier = Modifier.size(24.dp)
                        )
                    }

                    Spacer(modifier = Modifier.width(32.dp))

                    // 재생/일시정지 버튼 (그라데이션 효과)
                    val buttonBrush = Brush.linearGradient(
                        colors = if (mode == "focus") 
                            listOf(Indigo500, Indigo600) 
                        else 
                            listOf(Emerald500, Emerald600)
                    )
                    
                    Box(
                        modifier = Modifier
                            .size(88.dp)
                            .shadow(
                                elevation = 10.dp, 
                                shape = RoundedCornerShape(32.dp),
                                spotColor = primaryColor
                            )
                            .clip(RoundedCornerShape(32.dp))
                            .background(buttonBrush)
                            .clickable { isActive = !isActive },
                        contentAlignment = Alignment.Center
                    ) {
                        Icon(
                            imageVector = if (isActive) Icons.Rounded.Pause else Icons.Rounded.PlayArrow,
                            contentDescription = if (isActive) "Pause" else "Start",
                            tint = Color.White,
                            modifier = Modifier.size(40.dp)
                        )
                    }
                    
                    // 레이아웃 균형을 위한 투명 박스
                    Spacer(modifier = Modifier.width(32.dp))
                    Box(modifier = Modifier.size(56.dp))
                }

                // 하단 문구
                Text(
                    text = "Focus on what matters most",
                    color = Gray400,
                    fontSize = 12.sp,
                    fontWeight = FontWeight.Medium
                )
            }
        }
    }
}

// --- 커스텀 컴포넌트 ---

@Composable
fun ModeButton(
    text: String,
    icon: ImageVector,
    isSelected: Boolean,
    selectedColor: Color,
    onClick: () -> Unit
) {
    val backgroundColor by animateColorAsState(if (isSelected) Color.White else Color.Transparent, label = "btnBg")
    val contentColor by animateColorAsState(if (isSelected) selectedColor else Gray500, label = "btnContent")
    val shadowElevation = if (isSelected) 2.dp else 0.dp

    Surface(
        onClick = onClick,
        shape = RoundedCornerShape(50),
        color = backgroundColor,
        shadowElevation = shadowElevation,
        modifier = Modifier.width(100.dp)
    ) {
        Row(
            modifier = Modifier.padding(vertical = 10.dp),
            horizontalArrangement = Arrangement.Center,
            verticalAlignment = Alignment.CenterVertically
        ) {
            Icon(
                imageVector = icon,
                contentDescription = null,
                tint = contentColor,
                modifier = Modifier.size(16.dp)
            )
            Spacer(modifier = Modifier.width(6.dp))
            Text(
                text = text,
                fontSize = 14.sp,
                fontWeight = FontWeight.Bold,
                color = contentColor
            )
        }
    }
}

@Composable
fun CircularProgressBar(
    progress: Float, // 0.0 ~ 1.0
    color: Color,
    strokeWidth: Dp,
    modifier: Modifier = Modifier
) {
    Canvas(modifier = modifier) {
        val size = this.size
        val stroke = strokeWidth.toPx()
        
        // 배경 원
        drawCircle(
            color = Gray100, // 배경색 (아주 연한 회색)
            style = Stroke(width = stroke)
        )
        
        // 진행 원 (Arc)
        // -90도부터 시작 (12시 방향)
        drawArc(
            color = color,
            startAngle = -90f,
            sweepAngle = 360 * (1 - progress), // 남은 시간을 채우는 방식 (줄어드는 애니메이션)
            useCenter = false,
            style = Stroke(width = stroke, cap = StrokeCap.Round)
        )
    }
}
